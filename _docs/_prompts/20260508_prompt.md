[[관리자 메뉴 수동 실행
별도 UI 작업 없음 — MANAGED_UNITS 추가만으로 /pages/admin 의 systemd 탭 카드에 자동 노출. Start 버튼이 곧 수동 실행 (oneshot service 라 1회 즉시 fire).

운영기 배포 절차

    sudo cp deploy/systemd/investment-advisor-macro-observer.{service,timer} /etc/systemd/system/
    # 3개 플레이스홀더 치환: __INSTALL_DIR__ / __VENV_PYTHON__ / __SYSTEM_USER__
    sudo sed -i 's|__INSTALL_DIR__|/home/dzp/dzp-main/program/investment-advisor|g; s|__VENV_PYTHON__|/home/dzp/dzp-main/program/investment-advisor/venv/bin/python|g; s|__SYSTEM_USER__|dzp|g' /etc/systemd/system/investment-advisor-macro-observer.{service,timer}

    sudo visudo -f /etc/sudoers.d/investment-advisor-systemd  # README 의 갱신본 적용
    sudo systemctl daemon-reload
    sudo systemctl enable --now investment-advisor-macro-observer.timer
